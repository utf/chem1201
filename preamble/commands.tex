\newcommand{\image}[3][]{%
  \begin{figure}[H]%
    \centering%
    \includegraphics[height={#3}]{{#2}.eps}%
    \ifx&#1&
    \else
      \caption*{#1}
    \fi
  \end{figure}%
}
\newcommand{\img}[2][]{%
  \begin{figure}[H] %
    \centerline{ %
    % \vspace*{-2pt}
    \includegraphics{{#2}.eps} %
    }\leavevmode
    \ifx&#1&
    \else
      \caption*{#1}
      % \vspace*{-8pt}
    \fi
      % \vspace*{-10pt}
  \end{figure}%
}
